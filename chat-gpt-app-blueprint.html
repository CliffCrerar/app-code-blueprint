      body: JSON.stringify({
        prompt: `User: ${currentMessage}\nAI:`,
        max_tokens: 100,
        temperature: 0.7,
      }),
    });
    const data = await response.json();
    const generatedText = data.choices[0].text.trim();

    await addChatMessage(`User: ${currentMessage}\nAI: ${generatedText}`, 'user');
    setCurrentMessage('');
  }
};

export const addChatMessage = async (text: string, sender: string) => {
  const timestamp = new Date();
  await firebase.firestore().collection('chatMessages').add({ text, sender, timestamp });
};

export const getChatMessages = async () => {
  const querySnapshot = await firebase.firestore().collection('chatMessages').orderBy('timestamp').get();
  return querySnapshot.docs.map((doc) => ({ id: doc.id, ...doc.data() })) as ChatMessage[];
};

export interface ChatMessage {
  id: string;
  text: string;
  sender: string;
  timestamp: firebase.firestore.Timestamp;
}
